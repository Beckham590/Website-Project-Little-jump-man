import { useState, useEffect } from "react";

import { motion } from "framer-motion";


export default function JumpCounterGame() {

const [localJumps, setLocalJumps] = useState(0);

const [globalJumps, setGlobalJumps] = useState(0);

const [jumpHeight, setJumpHeight] = useState(0);


// Load global jumps from localStorage (simulating daily reset)

useEffect(() => {

const today = new Date().toDateString();

const storedData = JSON.parse(localStorage.getItem("jumpCounterData"));


if (storedData && storedData.date === today) {

setGlobalJumps(storedData.global);

} else {

localStorage.setItem(

"jumpCounterData",

JSON.stringify({ date: today, global: 0 })

);

}

}, []);


const handleJump = () => {

const newLocal = localJumps + 1;

setLocalJumps(newLocal);


const newGlobal = globalJumps + 1;

setGlobalJumps(newGlobal);


setJumpHeight(newLocal * 10); // progressively higher


// Save global count

const today = new Date().toDateString();

localStorage.setItem(

"jumpCounterData",

JSON.stringify({ date: today, global: newGlobal })

);

};


return (

<div className="min-h-screen bg-black text-white flex flex-col items-center justify-between font-mono">

{/* Title */}

<h1 className="text-6xl text-yellow-400 pixel-font mt-6">JUMP COUNTER</h1>


{/* Jump button */}

<button

onClick={handleJump}

className="bg-red-600 hover:bg-red-700 text-white text-4xl px-12 py-6 rounded-xl border-4 border-red-900 shadow-lg pixel-font"

>

JUMP

</button>


{/* Pixel Man */}

<div className="relative w-full flex justify-center h-64 overflow-hidden">

<motion.div

animate={{ y: -jumpHeight }}

transition={{ type: "spring", stiffness: 200, damping: 10 }}

className="w-12 h-12 bg-green-500 pixel-man"

/>

</div>


{/* Counters */}

<div className="bg-gray-900 w-full py-4 flex justify-around text-lg border-t-4 border-yellow-400">

<p className="pixel-font">Your Jumps: {localJumps}</p>

<p className="pixel-font">Global Jumps Today: {globalJumps}</p>

</div>


<style jsx>{`

.pixel-font {

font-family: 'Press Start 2P', cursive;

}

.pixel-man {

image-rendering: pixelated;

}

`}</style>

</div>

);

}
